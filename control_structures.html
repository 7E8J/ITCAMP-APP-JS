3. Control structures (if/else statements, loops)

ref: https://en.wikibooks.org/wiki/JavaScript/Control_structures

if / else

if ( condition ) {
  // block of statements
} else {
  // block of statements
}

Example: 

const a = 3;
const b = "3";
if (a == b) {
  alert("The two variables contains the same value, but may have different data types.");
} else {
  alert("The two variables contain different values.");
}

// an example without 'else'
const c = 6 / 2;
if (a === c) {
  alert("The two variables contains the same value and are of the same data type.");
}


Example 2: 

// same as above; but this abbreviated syntax is not recommended.

const a = 3;
const b = "3";

if (a == b) alert("The two variables contains the same value, but may have different data types.");
else alert("The two variables contain different values.");

const c = 6 / 2;
if (a === c) alert("The two variables contains the same value and are of the same data type.");

Switch

switch (variable) {
  case "A":
    // code
    break;
  case "B":
    // code
    break;
  default:   // analog to 'else' without any further 'if'
    // code
    break;
}

Example: 

let x = 1;

switch (x) {
  case 0:
    alert("Off")
    break;
  case 1:
    alert("On")
    break;
  default:
    alert("Error")
    break
}

Loop For

Instead of writing:

Helloworld0
Helloworld1
Helloworld2
Helloworld3
Helloworld4
Helloworld5
Helloworld6
Helloworld7
Helloworld8
Helloworld9

Just write:

for (let i = 0; i < 10; i++) {
  console.log("Helloworl" + i)
}

Loop For In

Instead of writing:

for (key in object) {
  // code block to be executed
}

Just write:

const person = {fname:"John", lname:"Doe", age:25};

let text = "";
for (let x in person) {
  text += person[x];
}
console.log(text)

Loop ForEach

const numbers = [45, 4, 9, 16, 25];

let txt = "";
numbers.forEach((value, index) => {
  txt += value;
});

console.log(txt)
